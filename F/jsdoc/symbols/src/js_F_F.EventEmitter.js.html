1 /**
  2  * Provides observer pattern for basic eventing. Directly uses Backbone.Events
  3  *
  4  * @class
  5  */
  6 F.EventEmitter = new Class(
  7 	Backbone.Events
  8 
  9 	/**
 10 		Bind one or more space separated events, or an events map,
 11 		to a `callback` function. Passing `"all"` will bind the callback to
 12 		all events fired.
 13 	
 14 		@name on
 15 		@memberOf F.EventEmitter.prototype
 16 		@function
 17 	*/
 18 	
 19 	/**
 20 		Bind events to only be triggered a single time. After the first time
 21 		the callback is invoked, it will be removed.
 22 		
 23 		@name once
 24 		@memberOf F.EventEmitter.prototype
 25 		@function
 26 	*/
 27 	
 28 	/**
 29 		Remove one or many callbacks. If `context` is null, removes all
 30 		callbacks with that function. If `callback` is null, removes all
 31 		callbacks for the event. If `events` is null, removes all bound
 32 		callbacks for all events.
 33 		
 34 		@name off
 35 		@memberOf F.EventEmitter.prototype
 36 		@function
 37 	*/
 38 	
 39 	/**
 40 		Trigger one or many events, firing all bound callbacks. Callbacks are
 41 		passed the same arguments as `trigger` is, apart from the event name
 42 		(unless you're listening on `"all"`, which will cause your callback to
 43 		receive the true name of the event as the first argument).
 44 		
 45 		@name trigger
 46 		@memberOf F.EventEmitter.prototype
 47 		@function
 48 	*/
 49 	
 50 	/**
 51 		An inversion-of-control version of `on`. Tell *this* object to listen to
 52 		an event in another object ... keeping track of what it's listening to.
 53 		
 54 		@name listenTo
 55 		@memberOf F.EventEmitter.prototype
 56 		@function
 57 	*/
 58 
 59 	/**
 60 		Tell this object to stop listening to either specific events ... or
 61 		to every object it's currently listening to.
 62 	
 63 		@name stopListening
 64 		@memberOf F.EventEmitter.prototype
 65 		@function
 66 	*/
 67 );
