1 (function() {
  2 	
  3 	/* Views
  4 	*******************/
  5 	
  6 	// Available as F.ListComponent.prototype.View
  7 	var ListView = F.View.extend({
  8 		tagName: 'ul',
  9 
 10 		initialize: function(options) {
 11 			options = options || {};
 12 
 13 			// Clumsy backbone way of calling parent class' constructor
 14 			F.View.prototype.initialize.apply(this, arguments);
 15 
 16 			this.collection = options.collection;
 17 			this.ItemView = options.ItemView || this.ItemView;
 18 			this.ItemTemplate = options.ItemTemplate || this.ItemTemplate;
 19 		},
 20 
 21 		render: function() {
 22 			if (F.options.debug) {
 23 				console.log('%s: rendering list view...', this.component && this.component.toString() || 'List view');
 24 			}
 25 			
 26 			if (this.parent && !$(this.el.parentNode).is(this.parent))
 27 				$(this.parent).append(this.el);
 28 
 29 			// Clear the list
 30 			this.$el.children().remove();
 31 
 32 			// Add and render each list item
 33 			this.collection.each(function(model) {
 34 				var view = new this.ItemView({
 35 					model: model,
 36 					template: this.ItemTemplate,
 37 					component: this.component
 38 				});
 39 				view.render();
 40 
 41 				// Store model
 42 				view.$el.data('model', model);
 43 
 44 				// Add the list item to the List
 45 				this.$el.append(view.el);
 46 			}.bind(this));
 47 
 48 			// Store the last time this view was rendered
 49 			this.rendered = new Date().getTime();
 50 
 51 			return this;
 52 		}
 53 	});
 54 
 55 	// Available as F.ListComponent.prototype.ItemView
 56 	var ItemView = F.View.extend({
 57 		tagName: 'li',
 58 		className: 'listItem'
 59 	});
 60 
 61 	/* Component
 62 	*******************/
 63 	F.ListComponent = new Class(/** @lends F.ListComponent# */{
 64 		toString: 'ListComponent',
 65 		extend: F.CollectionComponent,
 66 	
 67 		/**
 68 		 * A component that can load and render a collection as a list
 69 		 *
 70 		 * @constructs
 71 		 * @extends F.CollectionComponent
 72 		 *
 73 		 * @param {Object} options							Options for this component and its view. Options not listed below will be passed to the view.
 74 		 * @param {Backbone.Collection} options.Collection	The collection class this list will be rendered from
 75 		 * @param {Backbone.View} options.ListView			The view class this list will be rendered with
 76 		 * @param {Template} [options.ListTemplate]			The template this list will be rendered with. Renders to a UL tag by default
 77 		 * @param {Backbone.View} options.ItemView			The view that individual items will be rendered with
 78 		 * @param {Template} options.ItemTemplate			The template that individual items will be rendered with
 79 		 *
 80 		 * @property {Backbone.Collection} Collection	The collection class this list will be rendered from
 81 		 * @property {Backbone.View} ListView			The view class this list will be rendered with
 82 		 * @property {Template} ListTemplate			The template this list will be rendered with. Renders to a UL tag by default
 83 		 * @property {Backbone.View} ItemView			The view that individual items will be rendered with
 84 		 * @property {Template} ItemTemplate			The template that individual items will be rendered with
 85 		 */
 86 		construct: function(options) {
 87 			// Set object properties from options object, removing them from the object thereafter
 88 			this.setPropsFromOptions(options, [
 89 				'Collection',
 90 				'ListTemplate',
 91 				'ListView',
 92 				'ItemTemplate',
 93 				'ItemView'
 94 			]);
 95 			
 96 			this.view = new this.ListView(_.extend({
 97 				component: this,
 98 				collection: this.collection,
 99 				ItemView: this.ItemView,
100 				ItemTemplate: this.ItemTemplate,
101 				events: {
102 					'click li': 'handleSelect'
103 				}
104 			}, options));
105 			
106 			this.selectedItem = null;
107 		},
108 	
109 		Collection: Backbone.Collection, // Collection component expects to have prototype.Collection or options.Collection
110 	
111 		ListTemplate: null,
112 		ListView: ListView,
113 	
114 		ItemTemplate: null,
115 		ItemView: ItemView,
116 	
117 		/**
118 		 * Handles item selection events
119 		 *
120 		 * @param {Event} evt	The jQuery event object
121 		 */
122 		handleSelect: function(evt) {
123 			var listItem = $(evt.currentTarget);
124 			var model = listItem.data('model');
125 			this.selectedItem = model.id;
126 		
127 			this.trigger('itemSelected', {
128 				listItem: listItem,
129 				model: model
130 			});
131 		}
132 	});
133 }());
