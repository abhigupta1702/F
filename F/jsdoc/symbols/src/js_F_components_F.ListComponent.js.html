1 (function() {
  2 	
  3 	/* Views
  4 	*******************/
  5 	
  6 	// Available as F.ListComponent.prototype.View
  7 	var ListView = F.View.extend({
  8 		tagName: 'ul',
  9 
 10 		initialize: function(options) {
 11 			options = options || {};
 12 
 13 			// Clumsy backbone way of calling parent class' constructor
 14 			F.View.prototype.initialize.apply(this, arguments);
 15 
 16 			this.collection = options.collection;
 17 			this.ItemView = options.ItemView || this.ItemView;
 18 			this.ItemTemplate = options.ItemTemplate || this.ItemTemplate;
 19 			
 20 			// Views array for subviews
 21 			this.subViews = [];
 22 			
 23 			// Clumsy backbone way of permabinding
 24 			this.addSubView = this.addSubView.bind(this);
 25 		},
 26 
 27 		addSubView: function(model) {
 28 			// Create view
 29 			var view = new this.ItemView({
 30 				model: model,
 31 				template: this.ItemTemplate,
 32 				component: this.component
 33 			});
 34 			view.render();
 35 			
 36 			// Add the list item to the List
 37 			this.$el.append(view.el);
 38 			
 39 			// Store the position in the views array
 40 			// Don't store the actual view to prevent circular references
 41 			view.$el.data('viewIndex', this.subViews.length);
 42 			
 43 			// Store in views array for removal later
 44 			this.subViews.push(view);
 45 		},
 46 		
 47 		remove: function() {
 48 			this.removeSubViews();
 49 			
 50 			F.View.prototype.remove.call(this, arguments);
 51 		},
 52 
 53 		removeSubView: function(modelOrViewIndex) {
 54 			var view = null;
 55 			var viewIndex = -1;
 56 			if (typeof viewIndex !== 'Number') {
 57 				_.some(this.subViews, function(tmpView, index) {
 58 					if (tmpView && tmpView.model === modelOrViewIndex) {
 59 						view = tmpView;
 60 						viewIndex = index;
 61 					}
 62 				}.bind(this));
 63 			}
 64 			else // get from view index
 65 				view = this.subViews[viewIndex];
 66 				
 67 			if (view) {
 68 				view.remove();
 69 				this.subViews[viewIndex] = undefined;
 70 			}
 71 		},
 72 		
 73 		removeSubViews: function() {
 74 			if (this.subViews.length) {
 75 				_.each(this.subViews, function(view) {
 76 					view.remove();
 77 				});
 78 				
 79 				this.subViews = [];
 80 			}
 81 		},
 82 
 83 		render: function() {
 84 			if (F.options.debug) {
 85 				console.log('%s: rendering list view...', this.component && this.component.toString() || 'List view');
 86 			}
 87 			
 88 			if (this.parent && !$(this.el.parentNode).is(this.parent))
 89 				$(this.parent).append(this.el);
 90 
 91 			// Remove previous views from the DOM
 92 			this.removeSubViews();
 93 			
 94 			// Add and render each list item
 95 			this.collection.each(this.addSubView);
 96 			
 97 			// Store the last time this view was rendered
 98 			this.rendered = new Date().getTime();
 99 
100 			this.trigger('renderComplete');
101 			
102 			return this;
103 		}
104 	});
105 
106 	// Available as F.ListComponent.prototype.ItemView
107 	var ItemView = F.View.extend({
108 		tagName: 'li',
109 		className: 'listItem'
110 	});
111 
112 	/* Component
113 	*******************/
114 	F.ListComponent = new Class(/** @lends F.ListComponent# */{
115 		toString: 'ListComponent',
116 		extend: F.CollectionComponent,
117 	
118 		/**
119 		 * A component that can load and render a collection as a list
120 		 *
121 		 * @constructs
122 		 * @extends F.CollectionComponent
123 		 *
124 		 * @param {Object} options							Options for this component and its view. Options not listed below will be passed to the view.
125 		 * @param {Backbone.Collection} options.Collection	The collection class this list will be rendered from
126 		 * @param {Backbone.View} options.ListView			The view class this list will be rendered with
127 		 * @param {Template} [options.ListTemplate]			The template this list will be rendered with. Renders to a UL tag by default
128 		 * @param {Backbone.View} options.ItemView			The view that individual items will be rendered with
129 		 * @param {Template} options.ItemTemplate			The template that individual items will be rendered with
130 		 *
131 		 * @property {Backbone.Collection} Collection	The collection class this list will be rendered from
132 		 * @property {Backbone.View} ListView			The view class this list will be rendered with
133 		 * @property {Template} ListTemplate			The template this list will be rendered with. Renders to a UL tag by default
134 		 * @property {Backbone.View} ItemView			The view that individual items will be rendered with
135 		 * @property {Template} ItemTemplate			The template that individual items will be rendered with
136 		 */
137 		construct: function(options) {
138 			// Set object properties from options object, removing them from the object thereafter
139 			this.setPropsFromOptions(options, [
140 				'Collection',
141 				'ListTemplate',
142 				'ListView',
143 				'ItemTemplate',
144 				'ItemView'
145 			]);
146 			
147 			this.view = new this.ListView(_.extend({
148 				component: this, // pass this as component so ItemView can trigger handleSelect if it likes
149 				collection: this.collection,
150 				ItemView: this.ItemView,
151 				ItemTemplate: this.ItemTemplate,
152 				events: {
153 					'click li': 'handleSelect'
154 				}
155 			}, options));
156 			
157 			this.selectedItem = null;
158 		},
159 	
160 		Collection: Backbone.Collection, // Collection component expects to have prototype.Collection or options.Collection
161 	
162 		ListTemplate: null,
163 		ListView: ListView,
164 	
165 		ItemTemplate: null,
166 		ItemView: ItemView,
167 	
168 		addModel: function(model) {
169 			// Add a subview for this model
170 			this.view.addSubView(model);
171 		},
172 		
173 		removeModel: function(model) {
174 			// Add a subview for this model
175 			this.view.removeSubView(model);			
176 		},
177 	
178 		/**
179 		 * Get the model associated with a list item
180 		 *
181 		 * @param {Node}	Node or jQuery Object to get model from
182 		 *
183 		 * @returns {Backbone.Model}	The model associated with the passed DOM element
184 		 */
185 		getModelFromLi: function(listItem) {
186 			return this.view.subViews[$(listItem).data('viewIndex')].model;
187 		},
188 	
189 		/**
190 		 * Get the view associated with a list item
191 		 *
192 		 * @param {Node}	Node or jQuery Object to get model from
193 		 *
194 		 * @returns {Backbone.View}	The view associated with the passed DOM element
195 		 */
196 		getViewFromLi: function(listItem) {
197 			return this.view.subViews[$(listItem).data('viewIndex')];
198 		},
199 		
200 		/**
201 		 * Handles item selection events
202 		 *
203 		 * @param {Event} evt	The jQuery event object
204 		 */
205 		handleSelect: function(evt) {
206 			// Get model from DOM el's data
207 			var model = this.getModelFromLi(evt.currentTarget);
208 			
209 			// Store ID of selected item
210 			this.selectedItem = model.id;
211 		
212 			this.trigger('itemSelected', {
213 				listItem: $(evt.currentTarget),
214 				model: model
215 			});
216 		}
217 	});
218 }());
