1 (function() {
  2 	
  3 	/* Views
  4 	*******************/
  5 	
  6 	// Available as F.ListComponent.prototype.View
  7 	var ListView = F.View.extend({
  8 		tagName: 'ul',
  9 
 10 		initialize: function(options) {
 11 			options = options || {};
 12 
 13 			// Clumsy backbone way of calling parent class' constructor
 14 			F.View.prototype.initialize.apply(this, arguments);
 15 
 16 			this.collection = options.collection;
 17 			this.ItemView = options.ItemView || this.ItemView;
 18 			this.ItemTemplate = options.ItemTemplate || this.ItemTemplate;
 19 			
 20 			// Views array for subviews
 21 			this.views = [];
 22 		},
 23 
 24 		remove: function() {
 25 			this.removeSubViews();
 26 			
 27 			F.View.prototype.remove.call(this, arguments);
 28 		},
 29 
 30 		removeSubViews: function() {
 31 			if (this.views.length) {
 32 				_.each(this.views, function(view) {
 33 					view.remove();
 34 				});
 35 				
 36 				this.views = [];
 37 			}
 38 		},
 39 
 40 		render: function() {
 41 			if (F.options.debug) {
 42 				console.log('%s: rendering list view...', this.component && this.component.toString() || 'List view');
 43 			}
 44 			
 45 			if (this.parent && !$(this.el.parentNode).is(this.parent))
 46 				$(this.parent).append(this.el);
 47 
 48 			// Remove previous views from the DOM
 49 			this.removeSubViews();
 50 			
 51 			// Add and render each list item
 52 			this.collection.each(function(model) {
 53 				var view = new this.ItemView({
 54 					model: model,
 55 					template: this.ItemTemplate,
 56 					component: this.component
 57 				});
 58 				view.render();
 59 				
 60 				// Store model
 61 				view.$el.data('model', model);
 62 				
 63 				// Add the list item to the List
 64 				this.$el.append(view.el);
 65 				
 66 				// Store in views array for removal later
 67 				this.views.push(view);
 68 			}.bind(this));
 69 			
 70 			// Store the last time this view was rendered
 71 			this.rendered = new Date().getTime();
 72 
 73 			_.defer(function() {
 74 				this.trigger('renderComplete');
 75 			}.bind(this));
 76 			
 77 			return this;
 78 		}
 79 	});
 80 
 81 	// Available as F.ListComponent.prototype.ItemView
 82 	var ItemView = F.View.extend({
 83 		tagName: 'li',
 84 		className: 'listItem'
 85 	});
 86 
 87 	/* Component
 88 	*******************/
 89 	F.ListComponent = new Class(/** @lends F.ListComponent# */{
 90 		toString: 'ListComponent',
 91 		extend: F.CollectionComponent,
 92 	
 93 		/**
 94 		 * A component that can load and render a collection as a list
 95 		 *
 96 		 * @constructs
 97 		 * @extends F.CollectionComponent
 98 		 *
 99 		 * @param {Object} options							Options for this component and its view. Options not listed below will be passed to the view.
100 		 * @param {Backbone.Collection} options.Collection	The collection class this list will be rendered from
101 		 * @param {Backbone.View} options.ListView			The view class this list will be rendered with
102 		 * @param {Template} [options.ListTemplate]			The template this list will be rendered with. Renders to a UL tag by default
103 		 * @param {Backbone.View} options.ItemView			The view that individual items will be rendered with
104 		 * @param {Template} options.ItemTemplate			The template that individual items will be rendered with
105 		 *
106 		 * @property {Backbone.Collection} Collection	The collection class this list will be rendered from
107 		 * @property {Backbone.View} ListView			The view class this list will be rendered with
108 		 * @property {Template} ListTemplate			The template this list will be rendered with. Renders to a UL tag by default
109 		 * @property {Backbone.View} ItemView			The view that individual items will be rendered with
110 		 * @property {Template} ItemTemplate			The template that individual items will be rendered with
111 		 */
112 		construct: function(options) {
113 			// Set object properties from options object, removing them from the object thereafter
114 			this.setPropsFromOptions(options, [
115 				'Collection',
116 				'ListTemplate',
117 				'ListView',
118 				'ItemTemplate',
119 				'ItemView'
120 			]);
121 			
122 			this.view = new this.ListView(_.extend({
123 				component: this, // pass this as component so ItemView can trigger handleSelect if it likes
124 				collection: this.collection,
125 				ItemView: this.ItemView,
126 				ItemTemplate: this.ItemTemplate,
127 				events: {
128 					'click li': 'handleSelect'
129 				}
130 			}, options));
131 			
132 			this.selectedItem = null;
133 		},
134 	
135 		Collection: Backbone.Collection, // Collection component expects to have prototype.Collection or options.Collection
136 	
137 		ListTemplate: null,
138 		ListView: ListView,
139 	
140 		ItemTemplate: null,
141 		ItemView: ItemView,
142 	
143 		/**
144 		 * Get the model associated with a list item
145 		 *
146 		 * @param {Node}	Node or jQuery Object to get model from
147 		 *
148 		 * @returns {Backbone.Model}	The model associated with the passed DOM element
149 		 */
150 		getModelFromLi: function(listItem) {
151 			return $(listItem).data('model');
152 		},
153 	
154 		/**
155 		 * Handles item selection events
156 		 *
157 		 * @param {Event} evt	The jQuery event object
158 		 */
159 		handleSelect: function(evt) {
160 			// Get model from DOM el's data
161 			var model = this.getModelFromLi(evt.currentTarget);
162 			
163 			// Store ID of selected item
164 			this.selectedItem = model.id;
165 		
166 			this.trigger('itemSelected', {
167 				listItem: $(evt.currentTarget),
168 				model: model
169 			});
170 		}
171 	});
172 }());
