  2 	function decapitalize(str) {
  3 		return str.slice(0, 1).toLowerCase()+str.slice(1);	
  4 	}

  6 	F.Component = new Class(/** @lends F.Component# */{
  7 		toString: 'Component',
  8 		extend: F.EventEmitter,
  9 		/**
 10 		 * Generic component class
 11 		 *
 12 		 * @extends F.EventEmitter
 13 		 * @constructs
 14 		 *
 15 		 * @param {Object} options	Component options
 16 		 * @param {Boolean} options.singly		Whether this component will allow multiple sub-components to be visible at once. If true, only one component will be visible at a time.
 17 		 * @param {Boolean} options.visible		If true, this component will be visible immediately.
 18 		 *
 19 		 * @property {Object} options	Default options for this component. These will be merged with options passed to the constructor.
 20 		 */
 21 		construct: function(options) {
 22 			// Looks funny, but it modifies options with defaults and makes them available to other constructors
 23 			this.mergeOptions({
 24 				singly: false, // Show only one subcomponent at a time
 25 				visible: false // Visible immediately or not
 26 			}, options);
 27 			
 28 			// Store options into object
 29 			// TBD: figure out what to do with these props if set on the object already
 30 			this.setPropsFromOptions(options, [
 31 				'singly', 
 32 				'visible'
 33 			]);
 34 			
 35 			// Sub components
 36 			this.components = {};
 37 			
 38 			// Hold the bubbled event listeners
 39 			this._bubbledEvts = {};
 40 			
 41 			// Make sure the following functions are always called in scope
 42 			// They are used in event handlers, and we want to be able to remove them
 43 			this.bind(this._setCurrentComponent);
 44 			this.bind(this.render);
 45 		},
 46 		
 47 		/**
 48 		 * Destroy this instance and free associated memory
 49 		 */
 50 		destruct: function() {
 51 			// If this module has a view in this.view, destroy it automatically
 52 			if (this.view)
 53 				this.view.remove();
 54 			
 55 			// Destroy sub-components
 56 			for (var component in this.components) {
 57 				this.components[component].destruct();
 58 				delete this[component];
 59 			}
 60 		
 61 			// Clear references to components
 62 			delete this.components;
 63 		},
 64 		
 65 		/**
 66 		 * Render the view associated with this component, if it has one
 67 		 *
 68 		 * @returns {F.Component}	this, chainable
 69 		 */
 70 		render: function() {
 71 			if (this.view) {
 72 				this.view.render();
 73 			}
 74 			
 75 			return this;
 76 		},
 77 	
 78 		/**
 79 		 * Set an event to bubble up the component chain by re-triggering it when the given sub-component triggers it
 80 		 * 
 81 		 * @param {String} componentName	Name of the component whose event to bubble
 82 		 * @param {String} evt				Name of event to bubble up
 83 		 *
 84 		 * @returns {F.Component}	this, chainable
 85 		 */
 86 		bubble: function(componentName, evt) {
 87 			if (!this[componentName]) {
 88 				console.error("%s: cannot set event '%s' for bubbling from component '%s', component does not exist", this.toString(), evt, componentName);
 89 				return this;
 90 			}
 91 			
 92 			if (!this._bubbledEvts[componentName])
 93 				this._bubbledEvts[componentName] = {};
 94 			
 95 			// Create a handler
 96 			var handler = this._bubbledEvts[componentName][evt] = function() {
 97 				// Turn the event arguments into an array
 98 				var args = Array.prototype.slice.call(arguments);
 99 				
100 				// Add the name of the event to the arguments array
101 				args.unshift(evt);
102 				
103 				// Call to bubble the event up
104 				this.trigger.apply(this, args);
105 			}.bind(this);
106 			
107 			// Add the listener
108 			this[componentName].on(evt, handler);
109 			
110 			return this;
111 		},
112 	
113 		/**
114 		 * Discontinue bubbling of a given event
115 		 * 
116 		 * @param {String} componentName	Name of the component whose event to stop bubbling
117 		 * @param {String} evt				Name of event that was set to bubble
118 		 *
119 		 * @returns {F.Component}	this, chainable
120 		 */
121 		unbubble: function(componentName, evt) {
122 			if (!this._bubbledEvts[componentName] || !this._bubbledEvts[componentName][evt]) {
123 				console.warn("%s: cannot discontinue bubbling of event '%s' for component '%s', event was not set for bubbling", this.toString(), evt, componentName);
124 				return this;
125 			}
126 
127 			// Remove the listener
128 			this[componentName].off(evt, this._bubbledEvts[componentName][evt]);
129 
130 			return this;
131 		},

133 		/**
134 		 * Add an instance of another component as a sub-component.
135 		 *
136 		 * this[subComponent.toString()] is used to reference the sub-component:
137 		 * 
138 		 *   this.List.show();
139 		 * 
140 		 * You can give a component an optional custom name as the second argument, then reference as such:
141 		 * 
142 		 *  this.myCustomComponent.show();
143 		 * 
144 		 * @param {F.Component} component	Instance of component
145 		 * @param {Function} componentName	Optional custom name for this component
146 		 *
147 		 * @returns {F.Component}	The sub-component that was added
148 		 */
149 		addComponent: function(component, componentName) {
150 			// Get the name of the component
151 			if (componentName) {
152 				// Tell component its new name, if provided
153 				componentName = decapitalize(componentName);
154 			}
155 			else {
156 				// Use lowercase of toString
157 				componentName = decapitalize(component.toString());
158 			}
159 			
160 			// Give component its new name
161 			component.setName(componentName);
162 			
163 			// Store component
164 			this[componentName] = this.components[componentName] = component;
165 		
166 			// Hide view by default
167 			if (component.view) {
168 				if (component.view.el) {
169 					if (component.visible === true) {
170 						// Call show method so view is rendered
171 						console.log('calling show with silent', component.toString());
172 						component.show({ silent: true });
173 					}
174 					else {
175 						// Just hide the el
176 						component.view.$el.hide();
177 					}
178 				}
179 				else {
180 					console.warn('Component %s has a view without an element', componentName, component, component.view, component.view.options);
181 				}
182 			}
183 			
184 			// Show a sub-component when it shows one of it's sub-components
185 			component.on('component:shown', this._setCurrentComponent);
186 			
187 			return component;
188 		},
189 	
190 		/**
191 		 * Remove a sub-component
192 		 *
193 		 * @param {Function} componentName	Component name
194 		 *
195 		 * @returns {F.Component}	this, chainable
196 		 */
197 		removeComponent: function(componentName) {
198 			var component = this[componentName];
199 		
200 			if (component !== undefined) {
201 				component.off('component:shown', this._setCurrentComponent);
202 		
203 				delete this[componentName];
204 				delete this.components[componentName];
205 			}
206 		
207 			return this;
208 		},

211 		/**
212 		 * Handles showing/hiding components in singly mode
213 		 *
214 		 * @param {Function} componentName	Component name
215 		 */
216 		_setCurrentComponent: function(componentName) {
217 			var newComponent = this.components[componentName];
218 		
219 			if (newComponent !== undefined) {
220 				// hide current component(s) for non-overlays
221 				if (this.singly && !newComponent.overlay) {
222 					this.hideComponents();
223 				}
224 			
225 				// Show self
226 				this.show();
227 			}
228 		},
229 	
230 		/**
231 		 * Show this component and emit an event so parent components can show themselves. Use options.silent to prevent component:shown event from firing
232 		 *
233 		 * @param {Object} options	Options object
234 		 *
235 		 * @returns {F.Component}	this, chainable
236 		 */
237 		show: function(options) {
238 			options = options || {};
239 			
240 			// Debug output
241 			if (F.options.debug) {
242 				// Don't show if already shown
243 				if (this.visible) {		
244 					console.log('%s: not showing self; already visible', this.toString());
245 				}
246 				else
247 					console.log('%s: showing self', this.toString());
248 			}
249 		
250 			if (!options.silent) {
251 				// Always trigger event before we show ourself so others can hide/show
252 				this.trigger('component:shown', this.toString(), this);	
253 			}
254 		
255 			// Always call show on the view so it has a chance to re-render
256 			if (this.view) {
257 				this.view.show();
258 			}
259 		
260 			this.visible = true;
261 	
262 			return this;
263 		},
264 	
265 		/**
266 		 * Hide this component
267 		 *
268 		 * @returns {F.Component}	this, chainable
269 		 */
270 		hide: function(options) {
271 			options = options || {};
272 			
273 			if (!this.visible)
274 				return false;
275 			
276 			if (F.options.debug) {
277 				console.log('%s: hiding self', this.toString());
278 			}
279 			
280 			// Hide the view
281 			if (this.view)
282 				this.view.hide();
283 			
284 			if (!options.silent) {
285 				// Trigger event after we hide ourself so we're out of the way before the next action
286 				this.trigger('component:hidden', this.toString(), this);
287 			}
288 		
289 			this.visible = false;
290 	
291 			return this;
292 		},

317 		 *

319 		 *
320 		 * @returns {F.Component}	this, chainable
321 		 */
322 		hideComponent: function(componentName) {
323 			var component = this.components[componentName];
324 			if (component !== undefined) {
325 				if (component.isVisible()) {
326 					// hide the component's element
327 					component.hide();
328 				}
329 			}
330 			else {
331 				console.warn(this.toString()+': cannot hide component %s, component not found', componentName);
332 			}
333 		
334 			return this;
335 		},

337 		/**
338 		 * Show a sub-component of this component by name.
339 		 *
340 		 * @param {Function} componentName	Component name
341 		 *
342 		 * @returns {F.Component}	this, chainable
343 		 */
344 		showComponent: function(componentName) {
345 			// Show the sub section, if not already showing
346 			var newComponent = this.components[componentName];
347 		
348 			if (newComponent !== undefined) {
349 				if (!newComponent.isVisible()) {
350 					// Hide the old component and show ourselves
351 					this._setCurrentComponent(componentName);
352 			
353 					// Show new component
354 					newComponent.show();
355 				}
356 				else {
357 					console.log(this.toString()+': not showing component %s, already visible', componentName);
358 				}
359 			}
360 			else {
361 				console.warn(this.toString()+': Cannot show component "'+componentName+'", not found');
362 			}
363 		
364 			return this;
365 		},

367 		/**
368 		 * Set a custom name for this component. Only useful before passing to addComponent

371 		 *
372 		 * @returns {F.Component}	this, chainable
373 		 */
374 		setName: function(customName) {
375 			this.toString = function() {
376 				return customName;
377 			};
378 			
379 			return this;

382 		/**
383 		 * Set properties of this instance from an options object, then remove the properties from the options object
384 		 *
385 		 * @param {Object} options	Options object with many properties
386 		 * @param {Array} props		Properties to copy from options object
387 		 *
388 		 * @returns {F.Component}	this, chainable
389 		 */
390 		setPropsFromOptions: function(options, props) {
391 			_.each(props, function(prop) {
392 				// Add the property to this instance, or use existing property if it's already there
393 				this[prop] = options[prop] || this[prop];
394 				// Delete the property from the options object
395 				delete options[prop];
396 			}.bind(this));
397 			
398 			return this;
399 		},
400 		
401 		/**
402 		 * Merges options in the following order:
403 		 *   Instance Options
404 		 *   Class options
405 		 *   Class defaults
406 		 *
407 		 * @param {Object} defaults	Default options object
408 		 * @param {Object} options	Instance options object (argument to constructor)
409 		 *
410 		 * @returns {Object}	Merged options object
411 		 */
412 		mergeOptions: function(defaults, options) {
413 			_.extend(
414 				options, 
415 				_.extend({}, defaults || {}, this.options || {}, options)
416 			);
417 			
418 			return options;
419 		}
