 16 		/**
 17 		 * Generic component class
 18 		 *
 19 		 * @extends F.EventEmitter
 20 		 * @constructs
 21 		 *
 22 		 * @param {Object} options	Component options
 23 		 * @param {Boolean} options.singly		Whether this component will allow multiple sub-components to be visible at once. If true, only one component will be visible at a time.
 24 		 * @param {Boolean} options.visible		If true, this component will be visible immediately.
 25 		 * @param {Boolean} options.debug		If true, show debug messages for this component.
 26 		 *
 27 		 * @property {Object} options			Default options for this component. These will be merged with options passed to the constructor.
 28 		 */
 29 		construct: function(options) {
 30 			// Merge options up the prototype chain
 31 			this.mergeOptions();
 32 			
 33 			// Add defaults to options arg and make available to other constructors
 34 			// Modify this.options to reflect any modifications the options arg may have made
 35 			this.applyOptions(options);
 36 			
 37 			// Sub components
 38 			this.components = {};
 39 			
 40 			// Hold the bubbled event listeners
 41 			this._bubbledEvts = {};
 42 			
 43 			// Make sure the following functions are always called in scope
 44 			// They are used in event handlers, and we want to be able to remove them
 45 			this.bind(this._handleSubComponentShown);
 46 			this.bind(this.render);

 49 		constructed: function() {
 50 			// Hide view by default
 51 			if (this.view) {
 52 				if (this.view.el) {
 53 					if (this.options.visible === true) {
 54 						// Call show method so view is rendered
 55 						this.show({ silent: true });
 56 					}
 57 					else {
 58 						// Just hide the el
 59 						this.view.$el.hide();
 60 					}
 61 				}
 62 				else {
 63 					console.warn('Component %s has a view without an element', this.toString, this, this.view, this.view.options);
 64 				}
 65 				
 66 				var self = this;
 67 				this.listenTo(this.view, 'renderComplete', function() {
 68 					if (typeof this.handleRenderComplete === 'function')
 69 						this.handleRenderComplete();
 70 					
 71 					self.trigger('view:rendered', {
 72 						component: self,
 73 						view: self.view
 74 					});
 75 				});
 76 			}

110 		/**
111 		 * Set an event to bubble up the component chain by re-triggering it when the given sub-component triggers it
112 		 * 
113 		 * @param {String} componentName	Name of the component whose event to bubble
114 		 * @param {String} evt				Name of event to bubble up
115 		 *
116 		 * @returns {F.Component}	this, chainable
117 		 */
118 		bubble: function(componentName, evt) {
119 			if (!this[componentName]) {
120 				console.error("%s: cannot set event '%s' for bubbling from component '%s', component does not exist", this.toString(), evt, componentName);
121 				return this;
122 			}
123 			
124 			if (!this._bubbledEvts[componentName])
125 				this._bubbledEvts[componentName] = {};
126 			
127 			// Create a handler
128 			var handler = this._bubbledEvts[componentName][evt] = function() {
129 				// Turn the event arguments into an array
130 				var args = Array.prototype.slice.call(arguments);
131 				
132 				// Add the name of the event to the arguments array
133 				args.unshift(evt);
134 				
135 				// Call to bubble the event up
136 				this.trigger.apply(this, args);
137 			}.bind(this);
138 			
139 			// Add the listener
140 			this[componentName].on(evt, handler);
141 			
142 			return this;
143 		},

145 		/**
146 		 * Discontinue bubbling of a given event
147 		 * 
148 		 * @param {String} componentName	Name of the component whose event to stop bubbling
149 		 * @param {String} evt				Name of event that was set to bubble
150 		 *
151 		 * @returns {F.Component}	this, chainable
152 		 */
153 		unbubble: function(componentName, evt) {
154 			if (!this._bubbledEvts[componentName] || !this._bubbledEvts[componentName][evt]) {
155 				console.warn("%s: cannot discontinue bubbling of event '%s' for component '%s', event was not set for bubbling", this.toString(), evt, componentName);
156 				return this;
157 			}
158 
159 			// Remove the listener
160 			this[componentName].off(evt, this._bubbledEvts[componentName][evt]);
161 
162 			return this;
163 		},

165 		/**
166 		 * Add an instance of another component as a sub-component.
167 		 *
168 		 * this[subComponent.toString()] is used to reference the sub-component:
169 		 * 
170 		 *   this.List.show();
171 		 * 
172 		 * You can give a component an optional custom name as the second argument, then reference as such:
173 		 * 
174 		 *  this.myCustomComponent.show();
175 		 * 
176 		 * @param {F.Component} component	Instance of component
177 		 * @param {Function} componentName	Optional custom name for this component
178 		 *
179 		 * @returns {F.Component}	The sub-component that was added
180 		 */
181 		addComponent: function(component, componentName) {
182 			// Get the name of the component
183 			if (componentName) {
184 				// Tell component its new name, if provided
185 				componentName = decapitalize(componentName);
186 			}
187 			else {
188 				// Use lowercase of toString
189 				componentName = decapitalize(component.toString());
190 			}
191 			
192 			// Give component its new name
193 			component.setName(componentName);
194 			
195 			// Store component
196 			this[componentName] = this.components[componentName] = component;
197 		
198 			// Store this component as the parent
199 			component.parent = this;
200 			
201 			// Show a sub-component when it shows one of it's sub-components
202 			this.listenTo(component, 'component:shown', this._handleSubComponentShown);
203 			
204 			return component;
205 		},

207 		/**
208 		 * Remove and destroy a sub-component
209 		 *
210 		 * @param {Function} componentName	Component name
211 		 *
212 		 * @returns {F.Component}	this, chainable
213 		 */
214 		removeComponent: function(componentName) {
215 			var component = this[componentName];
216 		
217 			if (component !== undefined) {
218 				this.stopListening(component);
219 		
220 				delete this[componentName];
221 				delete this.components[componentName];
222 				
223 				if (typeof component.destruct === 'function')
224 					component.destruct();
225 			}
226 		
227 			return this;
228 		},

263 		/**
264 		 * Show this component and emit an event so parent components can show themselves. Use options.silent to prevent component:shown event from firing
265 		 *
266 		 * @param {Object} options	Options object
267 		 *
268 		 * @returns {F.Component}	this, chainable
269 		 */
270 		show: function(options) {
271 			options = options || {};
272 			
273 			// Debug output
274 			if (this.inDebugMode()) {
275 				// Don't show if already shown
276 				if (this.isVisible())
277 					console.log('%s: not showing self; already visible', this.toString());
278 				else
279 					console.log('%s: showing self', this.toString());
280 			}
281 		
282 			if (!options.silent && !this.options.independent) {
283 				// Always trigger event before we show ourself so others can hide/show
284 				this.trigger('component:shown', {
285 					name: this.toString(),
286 					component: this
287 				});
288 			}
289 		
290 			// Always call show on the view so it has a chance to re-render
291 			if (this.view) {
292 				this.view.show();
293 			}
294 			
295 			// Call setup if we're not setup
296 			if (!this.options.isSetup && typeof this.setup === 'function') {
297 				this.setup(options);
298 				this.options.isSetup = true;
299 			}
300 			
301 			this.options.visible = true;
302 	
303 			return this;
304 		},

306 		/**
307 		 * Hide this component
308 		 *
309 		 * @returns {F.Component}	this, chainable
310 		 */
311 		hide: function(options) {
312 			options = options || {};
313 			
314 			if (!this.isVisible())
315 				return false;
316 			
317 			if (this.inDebugMode()) {
318 				console.log('%s: hiding self', this.toString());
319 			}
320 			
321 			// Hide the view
322 			if (this.view)
323 				this.view.hide();
324 			
325 			if (!options.silent) {
326 				// Trigger event after we hide ourself so we're out of the way before the next action
327 				this.trigger('component:hidden', {
328 					name: this.toString(),
329 					component: this
330 				});
331 			}
332 			
333 			// Call teardown if we're setup
334 			if (this.options.isSetup && typeof this.teardown === 'function') {
335 				this.teardown(options);
336 				this.options.isSetup = false;
337 			}
338 		
339 			// Hide children
340 			if (this.options.hideChildren !== false)
341 				this.hideAllSubComponents();
342 		
343 			this.options.visible = false;
344 	
345 			return this;
346 		},

441 		/**
442 		 * Merge options up the prototype chain. Options defined in the child class take precedence over those defined in the parent class.
443 		 */
444 		mergeOptions: function() {
445 			// Create a set of all options in the correct order
446 			var optionSets = [];
447 			var proto = this.constructor.prototype;
448 			while (proto) {
449 				if (proto.hasOwnProperty('options')) {
450 					optionSets.unshift(proto.options);
451 				}
452 				proto = proto.superClass;
453 			}
454 			
455 			// All options should end up merged into a new object
456 			// That is, move our reference out of the prototype before we modify it
457 			optionSets.unshift({});
458 			
459 			// Perform the merge and store the new options object
460 			this.options = _.extend.apply(_, optionSets);
461 		},

463 		/**
464 		 * Applies passed options to instance options and applies instance options to passed options.
465 		 * Individual passed options will not be applied to instance options unless they are defined in default options for the class or parent class.
466 		 * <br>Note: The options merge is one level deep.
467 		 * <br>Note: This function assumes that <code>this.options</code> does not refer to an object in the prototype.
468 		 *
469 		 * @param {Object} options	Instance options object (usually argument to constructor)
470 		 *
471 		 * @returns {Object}	Merged options object
472 		 */
473 		applyOptions: function(options) {
474 			// Assume we already have moved our reference to this.options out of the prototype
475 			// Apply options from passed object to this.options
476 			for (var option in options) {
477 				if (this.options.hasOwnProperty(option))
478 					this.options[option] = options[option];
479 			}
480 			
481 			// Apply any missing defaults back to passed options object
482 			_.extend(options, this.options);
483 
484 			return options;
485 		}
