1 (function() {
  2 	
  3 	/* Views
  4 	*******************/
  5 	// Available as F.FormComponent.prototype.View
  6 	var FormView = F.View.extend(/** @lends F.FormComponent.prototype.View# */{
  7 		tagName: 'form',
  8 		events: {
  9 			'submit': 'handleSubmit'
 10 		}
 11 	});
 12 	
 13 	
 14 	/* Component
 15 	*******************/
 16 	F.FormComponent = new Class(/** @lends F.FormComponent# */{
 17 		toString: 'FormComponent',
 18 		extend: F.ModelComponent,
 19 	
 20 		/**
 21 		 * A component that can display an add/edit form for a model and handle form submission and save events
 22 		 *
 23 		 * @constructs
 24 		 * @extends F.ModelComponent
 25 		 *
 26 		 * @param {Object} options					Options for this component and its view. Options not listed below will be passed to the view.
 27 		 * @param {Backbone.Model} options.Model	The model class that the form will manipulate. Not an instance of the model, but the model class itself
 28 		 * @param {Backbone.View} options.View		The view class that the form will be rendered to
 29 		 * @param {Template} options.Template		The template that the form will be rendered with
 30 		 *
 31 		 * @property {Backbone.Model} Model	The model class that the form will manipulate. Not an instance of the model, but the model class itself
 32 		 * @property {Backbone.View} View	The view class that the form will be rendered to
 33 		 * @property {Template} Template	The template that the form will be rendered with
 34 		 */
 35 		construct: function(options) {
 36 			this.setPropsFromOptions(options, [
 37 				'Model',
 38 				'View',
 39 				'Template'
 40 			]);
 41 		
 42 			// Create a new edit view that responds to submit events
 43 			this.view = new this.View(_.extend({
 44 				component: this,
 45 				template: this.Template
 46 			}, options));
 47 		
 48 			// Create a blank model
 49 			this.model = new this.Model();
 50 		},
 51 	
 52 		View: FormView,
 53 	
 54 		Template: null,
 55 
 56 		/**
 57 		 * Clears the form by rendering it with a new, empty model
 58 		 */
 59 		clear: function() {
 60 			// Create a new model instead of resetting the old one
 61 			this.model = new this.Model();
 62 
 63 			// Render the view so it will be blank again
 64 			this.render();
 65 		
 66 			return this;
 67 		},
 68 	
 69 		/**
 70 		 * Handles form submit events
 71 		 *
 72 		 * @param {Event} evt	The jQuery event object
 73 		 */
 74 		handleSubmit: function(evt) {
 75 			// Get the data from the form fields
 76 			var fields = this.view.$el.serializeArray();
 77 		
 78 			// Build a data object from fields
 79 			var data = {};
 80 			_.each(fields, function(field) {
 81 				data[field.name] = field.value;
 82 			});
 83 		
 84 			// Perform the save, passing our modified as the second arg
 85 			this.save(data);
 86 
 87 			// Since this is a DOM event handler, prevent form submission
 88 			return false;
 89 		}
 90 	});
 91 
 92 }());
