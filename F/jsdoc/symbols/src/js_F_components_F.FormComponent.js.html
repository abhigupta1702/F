1 (function() {
  2 	
  3 	/* Views
  4 	*******************/
  5 	// Available as F.FormComponent.prototype.View
  6 	var FormView = F.View.extend(/** @lends F.FormComponent.prototype.View# */{
  7 		tagName: 'form',
  8 		events: {
  9 			'submit': 'handleSubmit'
 10 		}
 11 	});
 12 	
 13 	
 14 	/* Component
 15 	*******************/
 16 	F.FormComponent = new Class(/** @lends F.FormComponent# */{
 17 		toString: 'FormComponent',
 18 		extend: F.ModelComponent,
 19 	
 20 		/**
 21 		 * A component that can display an add/edit form for a model and handle form submission and save events
 22 		 *
 23 		 * @constructs
 24 		 * @extends F.ModelComponent
 25 		 *
 26 		 * @param {Object} options			Options for this component and its view. Options not listed below will be passed to the view.
 27 		 *
 28 		 * @property {Backbone.Model} Model	The model class that the form will manipulate. Not an instance of the model, but the model class itself
 29 		 * @property {Backbone.View} View	The view class that the form will be rendered to
 30 		 * @property {Template} Template	The template that the form will be rendered with
 31 		 */
 32 		construct: function(options) {
 33 			// Create a new edit view that responds to submit events
 34 			this.view = new this.View(_.extend({
 35 				component: this,
 36 				template: this.Template
 37 			}, options));
 38 		
 39 			// Create a blank model
 40 			this.model = new this.Model();
 41 			
 42 			this.bind(this.handleSubmit);
 43 		},
 44 
 45 		View: FormView,
 46 	
 47 		Template: null,
 48 
 49 		/**
 50 		 * Clears the form by rendering it with a new, empty model
 51 		 *
 52 		 * @returns {F.Component}	this, chainable
 53 		 */
 54 		clear: function() {
 55 			// Create a new model instead of resetting the old one
 56 			this._setModel(new this.Model());
 57 
 58 			// Call the load success function
 59 			var trigger = true;
 60 			if (typeof this.handleLoadSuccess === 'function')
 61 				this.handleLoadSuccess(this.model);
 62 			
 63 			// Render the view so it will be blank again
 64 			this.render();
 65 		
 66 			return this;
 67 		},
 68 		
 69 		/**
 70 		 * Blurs focus from the form, mostly for iOS
 71 		 */
 72 		doBlur: function() {
 73 			// Blur focus to the submit button in order to hide keyboard on iOS
 74 			// This won't work for every situation, such as forms that don't have submit buttons
 75 			var $button = this.view.$('[type="submit"], button').filter(':visible').last().focus();
 76 			setTimeout(function() {
 77 				$button.blur();
 78 			}, 10);
 79 		},
 80 	
 81 		/**
 82 		 * Handles form submit events
 83 		 *
 84 		 * @param {Event} evt	The jQuery event object
 85 		 */
 86 		handleSubmit: function(evt) {
 87 			this.doBlur();
 88 			
 89 			// Since this is a DOM event handler, prevent form submission
 90 			if (evt && evt.preventDefault)
 91 				evt.preventDefault();
 92 			
 93 			this.saveForm();
 94 		},
 95 		
 96 		/**
 97 		 * Read data from the form. Override this function customization of extracting your form data
 98 		 *
 99 		 * @returns {Object}	Data read from form
100 		 */
101 		extractValuesFromForm: function() {
102 			// Get the data from the form fields
103 			var $form = this.view.$el;
104 			if (this.view.el.tagName !== 'FORM')
105 				$form = this.view.$('form');
106 				
107 			var fields = $form.filter(':not([data-serialize="false"])').serializeArray();
108 			
109 			// Build a data object from fields
110 			var data = {};
111 			_.each(fields, function(field) {
112 				F.set(data, field.name, field.value, true);
113 			});
114 			
115 			return data;
116 		},
117 		
118 		/**
119 		 * Read the data from the form and store it in the model
120 		 */
121 		setValuesFromForm: function() {
122 			this.model.set(this.extractValuesFromForm());
123 		},
124 		
125 		/**
126 		 * Read the data from the form and perform the save
127 		 *
128 		 * @param {Function} callback	A callback to execute when the save is complete
129 		 */
130 		saveForm: function(callback) {
131 			// Perform the save, passing our new, modified data
132 			this.save(this.extractValuesFromForm(), callback);
133 		}
134 		
135 	});
136 }());
