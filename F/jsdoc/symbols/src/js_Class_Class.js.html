1 /**
  2  * Crockford's new_constructor pattern, modified to allow walking the prototype chain, automatic init/destruct calling of super classes, and easy toString methods
  3  *
  4  * @param {Object} descriptor						Descriptor object
  5  * @param {String or Function} descriptor.toString	A string or method to use for the toString of this class and instances of this class
  6  * @param {Object} descriptor.extend				The class to extend
  7  * @param {Function} descriptor.construct			The constructor (setup) method for the new class
  8  * @param {Function} descriptor.destruct			The destructor (teardown) method for the new class
  9  * @param {Anything} descriptor.*					Other methods and properties for the new class
 10  *
 11  * @returns {Class} The created class.
 12 */
 13 function Class(descriptor) {
 14 	descriptor = descriptor || {};
 15 	
 16 	if (descriptor.hasOwnProperty('extend') && !descriptor.extend) {
 17 		console.warn('Class: %s is attempting to extend a non-truthy thing', descriptor.toString === 'function' ? descriptor.toString : descriptor.toString, descriptor.extend);
 18 	}
 19 	
 20 	// Extend Object by default
 21 	var extend = descriptor.extend || Object;
 22 
 23 	// Construct and destruct are not required
 24 	var construct = descriptor.construct;
 25 	var destruct = descriptor.destruct;
 26 
 27 	// Remove special methods and keywords from descriptor
 28 	delete descriptor.extend;
 29 	delete descriptor.destruct;
 30 	delete descriptor.construct;
 31 	
 32 	// Add toString method, if necessary
 33 	if (descriptor.hasOwnProperty('toString') && typeof descriptor.toString !== 'function') {
 34 		// Return the string provided
 35 		var classString = descriptor.toString;
 36 		descriptor.toString = function() {
 37 			return classString.toString();
 38 		};
 39 	}
 40 	else if (!descriptor.hasOwnProperty('toString') && extend.prototype.hasOwnProperty('toString')) {
 41 		// Use parent's toString
 42 		descriptor.toString = extend.prototype.toString;
 43 	}
 44 	
 45 	// The remaining properties in descriptor are our methods
 46 	var methodsAndProps = descriptor;
 47 	
 48 	// Create an object with the prototype of the class we're extending
 49 	var prototype = Object.create(extend && extend.prototype);
 50 	
 51 	// Store super class as a property of the new class' prototype
 52 	prototype.superClass = extend.prototype;
 53 	
 54 	// Copy new methods into prototype
 55 	if (methodsAndProps) {	
 56 		for (var key in methodsAndProps) {
 57 			if (methodsAndProps.hasOwnProperty(key)) {
 58 				prototype[key] = methodsAndProps[key];
 59 				
 60 				// Store the method name so calls to inherited() work
 61 				if (typeof methodsAndProps[key] === 'function') {
 62 					prototype[key]._methodName = key;
 63 					prototype[key]._parentProto = prototype;
 64 				}
 65 			}
 66 		}
 67 	}
 68 	
 69 	/**
 70 	 * A function that calls an inherited method by the same name as the callee
 71 	 *
 72 	 * @param {Arguments} args	Unadulterated arguments array from calling function
 73 	*/
 74 	prototype.inherited = function(args) {
 75 		// Get the function that call us from the passed arguments objected
 76 		var caller = args.callee;
 77 
 78 		// Get the name of the method that called us from a property of the method
 79 		var methodName = caller._methodName;
 80 		
 81 		if (!methodName) {
 82 			console.error("Class.inherited: can't call inherited method: calling method did not have _methodName", args.callee);
 83 			return;
 84 		}
 85 
 86 		// Start iterating at the prototype that this function is defined in
 87 		var curProto = caller._parentProto;
 88 		var inheritedFunc = null;
 89 		
 90 		// Iterate up the prototype chain until we find the inherited function
 91 		while (curProto.superClass) {
 92 			curProto = curProto.superClass;
 93 			inheritedFunc = curProto[methodName];
 94 			if (typeof inheritedFunc === 'function')
 95 				break;
 96 		}
 97 		
 98 		if (typeof inheritedFunc === 'function') {
 99 			// Store our inherited function
100 			var oldInherited = this.inherited;
101 			
102 			// Overwrite our inherited function with that of the prototype so the called function can call its parent
103 			this.inherited = curProto.inherited;
104 			
105 			// Call the inherited function our scope, apply the passed args array
106 			var retVal = inheritedFunc.apply(this, args);
107 			
108 			// Revert our inherited function to the old function
109 			this.inherited = oldInherited;
110 			
111 			// Return the value called by the inherited function
112 			return retVal;
113 		}
114 		else {
115 			console.warn("Class.inherited: can't call inherited method for '%s': no method by that name found", methodName);			
116 		}
117 	};
118 	
119 	/**
120 	 * Binds a method to the execution scope of this instance
121 	 *
122 	 * @param {Function} func	The this.method you want to bind
123 	 */
124 	prototype.bind = function(func) {
125 		// Bind the function to always execute in scope
126 		var boundFunc = func.bind(this);
127 		
128 		// Store the method name
129 		boundFunc._methodName = func._methodName;
130 		
131 		// Store the bound function back to the class
132 		this[boundFunc._methodName] = boundFunc;
133 		
134 		// Return the bound function
135 		return boundFunc;
136 	};
137 
138 	/**
139 	 * Call the destruct method of all inherited classes
140 	 */
141 	prototype.destruct = function() {
142 		// Call our destruct method first
143 		if (typeof destruct === 'function') {
144 			destruct.apply(this);
145 		}
146 		
147 		// Call superclass destruct method after this class' method
148 		if (extend && extend.prototype && typeof extend.prototype.destruct === 'function') {
149 			extend.prototype.destruct.apply(this);			
150 		}
151 	};
152 	
153 	/**
154 	 * Construct is called automatically
155 	 */
156 	// Create a chained construct function which calls the superclass' construct function
157 	prototype.construct = function() {
158 		// Add a blank object as the first arg to the constructor, if none provided
159 		var args = arguments; // get around JSHint complaining about modifying arguments
160 		if (args[0] === undefined) {
161 			args.length = 1;
162 			args[0] = {};
163 		}
164 		
165 		// call superclass constructor
166 		if (extend && extend.prototype && typeof extend.prototype.construct === 'function') {
167 			extend.prototype.construct.apply(this, arguments);			
168 		}
169 
170 		// call constructor
171 		if (typeof construct === 'function') {
172 			construct.apply(this, arguments);
173 		}
174 	};
175 	
176 	// Create a function that generates instances of our class and calls our construct functions
177 	/** @private */
178 	var instanceGenerator = function() {
179 		// Create a new object with the prototype we built
180 		var instance = Object.create(prototype);
181 		
182 		// Call all inherited construct functions
183 		prototype.construct.apply(instance, arguments);
184 		
185 		return instance;
186 	};
187 	
188 	// Set the prototype of our instance generator to the prototype of our new class so things like MyClass.prototype.method.apply(this) work
189 	instanceGenerator.prototype = prototype;
190 	
191 	// The constructor, as far as JS is concerned, is actually our instance generator
192 	prototype.constructor = instanceGenerator;
193 	
194 	return instanceGenerator;
195 }
196 
197 if (!Object.create) {
198 	/**
199 	 * Polyfill for Object.create. Creates a new object with the specified prototype.
200 	 * 
201 	 * @author <a href="https://developer.mozilla.org/en/JavaScript/Reference/Global_Objects/Object/create/">Mozilla MDN</a>
202 	 *
203 	 * @param {Object} prototype	The prototype to create a new object with
204 	 */
205 	Object.create = function (prototype) {
206 		if (arguments.length > 1) {
207 			throw new Error('Object.create implementation only accepts the first parameter.');
208 		}
209 		function Func() {}
210 		Func.prototype = prototype;
211 		return new Func();
212 	};
213 }
214 
215 if (!Function.prototype.bind) {
216 	/**
217 	 * Polyfill for Function.bind. Binds a function to always execute in a specific scope.
218 	 * 
219 	 * @author <a href="https://developer.mozilla.org/en/JavaScript/Reference/Global_Objects/Function/bind">Mozilla MDN</a>
220 	 *
221 	 * @param {Object} oThis	The scope to bind the function to
222 	 */
223   Function.prototype.bind = function (oThis) {
224 	if (typeof this !== "function") {
225 	  // closest thing possible to the ECMAScript 5 internal IsCallable function
226 	  throw new TypeError("Function.prototype.bind - what is trying to be bound is not callable");
227 	}
228 
229 	var aArgs = Array.prototype.slice.call(arguments, 1), 
230 		fToBind = this, 
231 		fNOP = function () {},
232 		fBound = function () {
233 		  return fToBind.apply(this instanceof fNOP
234 								 ? this
235 								 : oThis,
236 							   aArgs.concat(Array.prototype.slice.call(arguments)));
237 		};
238 
239 	fNOP.prototype = this.prototype;
240 	fBound.prototype = new fNOP();
241 
242 	return fBound;
243   };
244 }
