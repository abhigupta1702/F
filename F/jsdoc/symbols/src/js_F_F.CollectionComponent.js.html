1 F.CollectionComponent = new Class(/** @lends F.CollectionComponent# */{
  2 	toString: 'CollectionComponent',
  3 	extend: F.Component,
  4 	options: {
  5 		defaultParams: {}
  6 	},
  7 	
  8 	/**
  9 	 * A component that can load and render a collection
 10 	 *
 11 	 * @constructs
 12 	 * @extends F.Component
 13 	 *
 14 	 * @param {Object} options							Options for this component
 15 	 * @param {Object} [options.defaultParams]			Default parameters to use when fetching this collection
 16 	 *
 17 	 * @property {Object} defaultParams				Default parameters to send with fetches for this collection. Can be overridden at instantiation. Calls to load(fetchParams) will merge fetchParams with defaultParams.
 18 	 * @property {Backbone.Collection} Collection	The collection class to operate on. Not an instance of a collection, but the collection class itself.
 19 	 */
 20 	construct: function(options) {
 21 		// Bind for use as listeners
 22 		this.bind(this.addModel);
 23 		this.bind(this.removeModel);
 24 		this.bind(this.render);
 25 		
 26 		this._useCollection(new this.Collection());
 27 		
 28 		// Default parameters are the prototype params + options params
 29 		this.options.defaultParams = _.extend({}, this.options.defaultParams, options.defaultParams);
 30 		
 31 		// Parameters to send with the request: just copy the default params
 32 		this.params = _.extend({}, this.options.defaultParams);
 33 	
 34 		// Store if this collection has ever been loaded
 35 		this.collectionLoaded = false;
 36 	},
 37 	
 38 	destruct: function() {
 39 		this._releaseCollection();
 40 	},
 41 	
 42 	Collection: Backbone.Collection,
 43 	
 44 	_useCollection: function(collection) {
 45 		// Store collection
 46 		this.collection = collection;
 47 		
 48 		// Re-render when the collection is fetched, items are added or removed
 49 		this.listenTo(this.collection, 'add', this.addModel);
 50 		this.listenTo(this.collection, 'remove', this.removeModel);
 51 		this.listenTo(this.collection, 'loaded', this.render); // custom event we call after fetches
 52 		// this.listenTo(this.collection, 'change', this.render); // Don't re-render on change! let the sub-views do that
 53 	},
 54 	
 55 	_releaseCollection: function() {
 56 		// Unbind events
 57 		this.stopListening(this.collection);
 58 		
 59 		// Remove reference to collection
 60 		this.collection = null;
 61 	},
 62 	
 63 	/**
 64 	 * Refresh this collection with the last parameters used
 65 	 *
 66 	 * @param {Function} callback	Optional callback to execute on successful fetch
 67 	 *
 68 	 * @returns {F.CollectionComponent}	this, chainable
 69 	 */
 70 	refresh: function(callback) {
 71 		// Just fetch the collection with the current params
 72 		this.fetch(this.params, callback);
 73 		
 74 		return this;
 75 	},
 76 	
 77 	/**
 78 	 * Callback called when model is added to collection
 79 	 */
 80 	addModel: function(model) {},
 81 	
 82 	/**
 83 	 * Callback called when model is removed from collection
 84 	 */
 85 	removeModel: function(model) {},
 86 	
 87 	/**
 88 	 * Clear the parameters from the last fetch. Useful when using refresh() on a filtered list.
 89 	 *
 90 	 * @returns {F.CollectionComponent}	this, chainable
 91 	 */
 92 	clearParams: function() {
 93 		this.params = {};
 94 		
 95 		return this;
 96 	},
 97 	
 98 	/**
 99 	 * Fetch the collection by fetching it from the server
100 	 *
101 	 * @param {Object} [fetchParams]	Parameters to pass when fetching
102 	 * @param {Function} [callback]		Callback to execute on successful fetch
103 	 *
104 	 * @returns {F.CollectionComponent}	this, chainable
105 	 */
106 	fetch: function(fetchParams, callback) {
107 		// Combine new params, if any, with defaults and store, overwriting previous params
108 		if (fetchParams)
109 			this.params = _.extend({}, this.options.defaultParams, fetchParams);
110 		else // Overwrite old params with defaults and send a request with only default params
111 			this.params = _.extend({}, this.options.defaultParams);
112 		
113 		// Fetch collection contents
114 		this.collection.fetch({
115 			data: this.params,
116 			success: function() {
117 				// Collection event
118 				this.collection.trigger('loaded');
119 
120 				// Component event
121 				this.trigger('collection:loaded', this.collection);
122 				this.collectionLoaded = true;
123 				
124 				if (typeof callback === 'function')
125 					callback.call(this, this.collection);
126 			}.bind(this)
127 		});
128 		
129 		return this;
130 	},
131 	
132 	
133 	/**
134 	 * Load a Backbone.Collection directly or create a collection from an array of data
135 	 *
136 	 * @param {mixed} collectionOrData	Backbone.Collection to load or Array of Objects with data to create the collection from
137 	 *
138 	 * @returns {F.CollectionComponent}	this, chainable
139 	 */
140 	load: function(collectionOrData) {
141 		this._releaseCollection();
142 
143 		if (collectionOrData instanceof Backbone.Collection)
144 			this._useCollection(collectionOrData);
145 		else
146 			this._useCollection(new this.Collection(collectionOrData));
147 		
148 		return this;
149 	},
150 	
151 	
152 	/**
153 	 * Show this component. Provide options.params to fetch with new parameters. The collection will be fetched before showing if it hasn't already
154 	 *
155 	 * @param {Object} options	Pass fetch parameters with options.params
156 	 *
157 	 * @returns {F.CollectionComponent}	this, chainable
158 	 */
159 	show: function(options) {
160 		options = options || {};
161 		if (options.params) {
162 			// Fetch the collection from the server
163 			this.fetch(options.params, function() {
164 				this.show({
165 					silent: options.silent
166 				});
167 			});
168 		}
169 		else if (!this.collectionLoaded) {
170 			// Perform initial load
171 			this.refresh(function() {
172 				this.show({
173 					silent: options.silent
174 				}); // show when we're fully loaded
175 			});
176 		}
177 		else
178 			this.inherited(arguments);
179 		
180 		return this;
181 	}

185 	 * Triggered when the collection is loaded from the server
186 	 *
187 	 * @name F.CollectionComponent#collection:loaded
188 	 * @event
189 	 *
190 	 * @param {Backbone.Collection}	collection	The collection that was loaded
191 	 */
192 });
