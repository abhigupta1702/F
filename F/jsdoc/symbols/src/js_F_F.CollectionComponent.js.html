1 F.CollectionComponent = new Class(/** @lends F.CollectionComponent# */{
  2 	toString: 'CollectionComponent',
  3 	extend: F.Component,
  4 	
  5 	/**
  6 	 * A component that can load and render a collection
  7 	 *
  8 	 * @constructs
  9 	 * @extends F.Component
 10 	 *
 11 	 * @param {Object} options							Options for this component
 12 	 * @param {Backbone.Collection} options.Collection	The collection class this component should operate on. Sets this.Collection
 13 	 * @param {Object} [options.defaultParams ]			Default parameters to use when fetching this collection
 14 	 *
 15 	 * @property {Object} defaultParams				Default parameters to send with fetches for this collection. Can be overridden at instantiation. Calls to load(fetchParams) will merge fetchParams with defaultParams.
 16 	 * @property {Backbone.Collection} Collection	The collection class to operate on. Not an instance of a collection, but the collection class itself.
 17 	 */
 18 	construct: function(options) {
 19 		// Store the collection class
 20 		this.setPropsFromOptions(options, [
 21 			'Collection'
 22 		]);
 23 		
 24 		// Create a collection
 25 		this.collection = new this.Collection();
 26 		
 27 		// Re-render when the collection resets
 28 		this.collection.on('reset', this.render);
 29 		
 30 		// Default parameters are the prototype params + options params
 31 		this.defaultParams = _.extend({}, this.defaultParams, options.defaultParams);
 32 		
 33 		// Parameters to send with the request: just copy the default params
 34 		this.params = _.extend({}, this.defaultParams);
 35 	
 36 		// Store if this collection has ever been loaded
 37 		this.collectionLoaded = false;
 38 	},
 39 	
 40 	Collection: Backbone.Collection,
 41 	
 42 	/**
 43 	 * Get the collection associated with this component
 44 	 *
 45 	 * @returns {Backbone.Collection}	The collection associated with this component
 46 	 */
 47 	getCollection: function() {
 48 		return this.collection;
 49 	},
 50 	
 51 	/**
 52 	 * Refresh this collection with the last parameters used
 53 	 *
 54 	 * @param {Function} callback	Optional callback to execute on successful fetch
 55 	 *
 56 	 * @returns {F.CollectionComponent}	this, chainable
 57 	 */
 58 	refresh: function(callback) {
 59 		// Just load the colleciton with the current params
 60 		this.load(this.params, callback);
 61 		
 62 		return this;
 63 	},
 64 	
 65 	/**
 66 	 * Fetch the collection with optional 
 67 	 *
 68 	 * @param {Object} fetchParams	Optional parameters to pass when fetching
 69 	 * @param {Function} callback	Optional callback to execute on successful fetch
 70 	 *
 71 	 * @returns {F.CollectionComponent}	this, chainable
 72 	 */
 73 	load: function(fetchParams, callback) {
 74 		// Combine new params, if any, with defaults and store, overwriting previous params
 75 		if (fetchParams)
 76 			this.params = _.extend({}, this.defaultParams, fetchParams);
 77 		
 78 		// Fetch collection contents
 79 		this.collection.fetch({
 80 			data: this.params,
 81 			success: function() {
 82 				this.trigger('collectionLoaded');
 83 				this.collectionLoaded = true;
 84 				
 85 				if (typeof callback === 'function')
 86 					callback.call(this, this.collection);
 87 			}.bind(this)
 88 		});
 89 		
 90 		return this;
 91 	},
 92 	
 93 	/**
 94 	 * Show this component. Provide options.params to fetch with new parameters. The collection will be fetched before showing if it hasn't already
 95 	 *
 96 	 * @param {Object} options	Pass fetch parameters with options.params
 97 	 *
 98 	 * @returns {F.CollectionComponent}	this, chainable
 99 	 */
100 	show: function(options) {
101 		options = options || {};
102 		if (options.params) {
103 			// Load the collection by itemId
104 			this.load(options.params, function() {
105 				this.show();
106 			});
107 		}
108 		else if (!this.collectionLoaded) {
109 			// Perform initial load
110 			this.refresh(function() {
111 				this.show(); // show when we're fully loaded
112 			});
113 		}
114 		else
115 			this.inherited(arguments);
116 			
117 		return this;
118 	}
119 });
