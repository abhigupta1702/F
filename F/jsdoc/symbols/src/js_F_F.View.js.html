1 (function() {
  2 	// A couple functions required to override delegateEvents
  3 	var delegateEventSplitter = /^(\S+)\s*(.*)$/;
  4 	var getValue = function(object, prop) {
  5 		if (!(object && object[prop])) return null;
  6 		return _.isFunction(object[prop]) ? object[prop]() : object[prop];
  7 	};
  8 	
  9 	F.View = Backbone.View.extend(/** @lends F.View# */{
 10 		/**;
 11 		 * Generic view class. Provides rendering and templating based on a model, eventing based on a component, and element management based on a parent
 12 		 *
 13 		 * @constructs
 14 		 * @extends Backbone.View
 15 		 *
 16 		 * @param {Object} options	Options for this view
 17 		 * @param {Template} options.template	The template to render this view with
 18 		 * @param {Element} [options.el]		The element, jQuery selector, or jQuery object to render this view to. Should not be used with options.parent
 19 		 * @param {Element} [options.parent]	The element, jQuery selector, or jQuery object to insert this components element into. Should not be used with options.el
 20 		 * @param {Backbone.Model} [options.model]	Instance of a Backbone model to render this view from
 21 		 * @param {Component} [options.component]	The component that events should be delegated to
 22 		 * @param {Object} [options.events]		Backbone events object indicating events to listen for on this view
 23 		 *
 24 		 * @property {Template} template		The template to render this view with
 25 		 *
 26 		 */
 27 		initialize: function() {
 28 			if (this.template || this.options.template) {
 29 				if (F.options.precompiledTemplates)
 30 					this.template = this.template || this.options.template;
 31 				else // For pre-compiled templates
 32 					this.template = Handlebars.template(this.template || this.options.template);
 33 			}
 34 			
 35 			// Always call in our scope so parents can remove change listeners on models by referencing view.render
 36 			this.render = this.render.bind(this);
 37 			
 38 			if (this.options.el) {
 39 				// Make sure the element is of the right tag
 40 				var actualNodeName = this.$el[0].nodeName.toUpperCase();
 41 				var requiredNodeName =  this.tagName.toUpperCase();
 42 				
 43 				if (actualNodeName != requiredNodeName) {
 44 					throw new Error('View: cannot create view, incorrect tag provided. Expected '+requiredNodeName+', but got '+actualNodeName);
 45 				}
 46 			
 47 				// Add the CSS class if it doesn't have it
 48 				this.$el.addClass(this.className);
 49 			}
 50 			
 51 			// Store parent, if provided
 52 			this.parent = this.options.parent;
 53 			
 54 			// Store the model
 55 			this.model = this.options.model;
 56 			
 57 			// Store the controlling component
 58 			this.component = this.options.component;
 59 			
 60 			// Add events
 61 			if (this.options.events)
 62 				this.delegateEvents(this.options.events);
 63 
 64 			// Add change listeners to the model, but only if has an on method
 65 			if (this.model && this.model.on) {
 66 				this.model.on('change', this.render);
 67 				
 68 				// TBD: Should we do these?
 69 				// this.model.on('reset', function() {
 70 				// 	console.log("View caught model reset!");
 71 				// 	console.log("%s: Re-rendering view because model was reset!", this.component && this.component.toString() || 'Orphaned view');
 72 				// 	this.render();
 73 				// }.bind(this));
 74 				
 75 				// TBD: Should we do these?
 76 				// this.model.on('loaded', function() {
 77 				// 	console.log("%s: Re-rendering view because model was loaded!", this.component && this.component.toString() || 'Orphaned view');
 78 				// 	this.render();
 79 				// }.bind(this));
 80 			}
 81 			
 82 			this.rendered = null;
 83 		},
 84 		
 85 		/**
 86 		 * Get the number of milliseconds seconds since the view was last rendered
 87 		 *
 88 		 * @returns {number} Number of milliseconds since this view was rendered
 89 		 */
 90 		age: function() {
 91 			return this.rendered !== null ? new Date().getTime() - this.rendered : -1;
 92 		},
 93 		
 94 		/**
 95 		 * Remove this view from the DOM and stop listening to model change events
 96 		 */
 97 		remove: function() {
 98 			this.$el.remove();
 99 		
100 			// Remove change listener
101 			if (this.model)
102 				this.model.off('change', this.render);
103 		},
104 		
105 		/**
106 		 * Show the view. The view will be rendered before it is shown if it hasn't already been rendered
107 		 *
108 		 * @returns {F.View}	this, chainable
109 		 */
110 		show: function() {
111 			// Ensure the view is rendered
112 			if (this.template) {
113 				this.renderOnce();
114 			}
115 			
116 			// Show the view
117 			this.$el.show();
118 			
119 			return this;
120 		},
121 	
122 		/**
123 		 * Hide the view
124 		 *
125 		 * @returns {F.View}	this, chainable
126 		 */
127 		hide: function() {
128 			// Hide the view
129 			this.$el.hide();
130 		
131 			return this;
132 		},
133 	
134 		/**
135 		 * Render the view only if it has not been rendered before (or has been reset)
136 		 *
137 		 * @returns {F.View}	this, chainable
138 		 */
139 		renderOnce: function() {
140 			// Only render the view if it has never been rendered
141 			if (this.rendered === null) {
142 				this.render();
143 			}
144 		
145 			return this;
146 		},
147 
148 		/**
149 		 * Render the view
150 		 *
151 		 * @returns {F.View}	this, chainable
152 		 */
153 		render: function() {
154 			if (F.options.debug) {
155 				console.log('%s: Rendering view...', this.component && this.component.toString() || 'Orphaned view');
156 			}
157 			
158 			if (this.template) {
159 				// Render template
160 				
161 				// Use this view's model, or the model of the component it's part of
162 				var model = this.model || this.component && this.component.model;
163 				
164 				// First, see if the model exists. If so, see if it has toJSON. If so, use model.toJSON. Otherwise, if model exists, use model. Otherwise, use {}
165 				this.$el.html(this.template(model && model.toJSON && model.toJSON() || model || {}));
166 			}
167 			
168 			// Add to parent, if not already there
169 			if (this.parent && !$(this.el.parentNode).is(this.parent)) {
170 				$(this.parent).append(this.el);
171 			}
172 		
173 			// Store the last time this view was rendered
174 			this.rendered = new Date().getTime();
175 			
176 			// Notify render has completed
177 			this.trigger('renderComplete');
178 			
179 			return this;
180 		},
181 		
182 		/**
183 		 * Delegate events to this view. This overrides Backbone.View.delegateEvents and lets us specify an object to call methods on instead of the view
184 		 *
185 		 * @param {Events hash to delegate} events
186 		 *
187 		 * @returns {F.View}	this, chainable
188 		 */
189 		delegateEvents: function(events) {
190 			if (!(events || (events = getValue(this, 'events')))) return;
191 			this.undelegateEvents();
192 			for (var key in events) {
193 				var method = events[key];
194 				
195 				var base = this.component;
196 						
197 				if (!_.isFunction(method)) {
198 					if (this.component) {
199 						var parts = events[key].split('.');
200 						
201 						if (parts.length > 1) {
202 							method = this.component;
203 						
204 							for (var i = 0; i < parts.length; i++) {
205 								var part = parts[i];
206 							
207 								// console.log('Looking for part ', part, 'in ', method);
208 								base = method;
209 								method = method[part];
210 							
211 								if (!method) {
212 									// console.warn('Could not find method %s', key);
213 									break;
214 								}
215 							}
216 						}
217 						else
218 							method = this.component[events[key]];
219 					}
220 					else {
221 						method = this[events[key]];
222 					}
223 				}
224 				
225 				if (!method) throw new Error('Method "' + events[key] + '" does not exist');
226 				
227 				var match = key.match(delegateEventSplitter);
228 				var eventName = match[1], selector = match[2];
229 
230 				// Execute in the scope of the base
231 				// TBD: determine if we ought to execute in the scope of the view itself ever?
232 				//		or leave that up to implementors to pass a bound function
233 				method = this.component ? _.bind(method, base) : _.bind(method, this);
234 				
235 				eventName += '.delegateEvents' + this.cid;
236 				if (selector === '') {
237 					this.$el.bind(eventName, method);
238 				} else {
239 					this.$el.delegate(selector, eventName, method);
240 				}
241 			}
242 			
243 			return this;
244 		}
245 	});
246 }());
