1 F.ModelComponent = new Class(/** @lends F.ModelComponent# */{
  2 	toString: 'ModelComponent',
  3 	extend: F.Component,
  4 	/**
  5 	 * A component that can load and render a model
  6 	 *
  7 	 * @constructs
  8 	 * @extends F.Component
  9 	 *
 10 	 * @param {Object} options	Options for this component
 11 	 * @param {Object} options.Model	Model class this component will be operating on. Sets this.Model
 12 	 *
 13 	 * @property {Backbone.Model} Model		The model class to operate on. Not an instance of a model, but the model class itself.
 14 	 */
 15 	construct: function(options) {
 16 		this.Model = this.Model || options.Model;
 17 	},
 18 	
 19 	Model: Backbone.Model,
 20 	
 21 	/***
 22 	 * Refresh the model
 23 	 *
 24 	 * @param {Function} callback	Callback to call after successful refresh
 25 	 *
 26 	 * @returns {F.ModelComponent}	this, chainable
 27 	 */
 28 	refresh: function(callback) {
 29 		this.model.fetch({
 30 			success: function() {
 31 				// Trigger model event
 32 				this.model.trigger('loaded');
 33 				
 34 				// Trigger component event
 35 				this.trigger('model:loaded', this.model);
 36 				
 37 				if (typeof callback === 'function')
 38 					callback.call(this, this.model);
 39 			}.bind(this),
 40 			error: function(model, response) {
 41 				console.warn('%s: Error loading model', this.toString());
 42 				
 43 				this.trigger('model:loadFailed', {
 44 					response: response
 45 				});
 46 			}.bind(this)
 47 		});
 48 		
 49 		return this;
 50 	},
 51 	
 52 	/**
 53 	 * Use a different item model
 54 	 *
 55 	 * @param {Backbone.Model} model
 56 	 *
 57 	 * @returns {F.ModelComponent}	this, chainable
 58 	 */
 59 	_setModel: function(model) {
 60 		this.model = model;
 61 	
 62 		if (this.model && this.model.off && this.view) {
 63 			this.view.setModel(model);
 64 		}
 65 		
 66 		return this;
 67 	},
 68 		
 69 	/**
 70 	 * Fetch a model with the given ID
 71 	 *
 72 	 * @param {String} itemId		ID of the item to fetch
 73 	 * @param {Function} [callback]	Callback to execute on successful fetch
 74 	 *
 75 	 * @returns {F.ModelComponent}	this, chainable
 76 	 */
 77 	fetch: function(itemId, callback) {
 78 		var data = {};
 79 		if (itemId !== undefined) { // add the ID passed to the model data
 80 			data[this.Model.prototype.idAttribute] = itemId;
 81 		}
 82 		
 83 		// Create a blank model
 84 		var model = new this.Model(data);
 85 	
 86 		// Fetch model contents
 87 		model.fetch({
 88 			// TBD: add fetch options?
 89 			success: function() {
 90 				// Assign the model to the view
 91 				this._setModel(model);
 92 				
 93 				// Notify
 94 				this.trigger('model:loaded', this.model);
 95 				
 96 				// Call callback
 97 				if (typeof callback === 'function')
 98 					callback.call(this, model);
 99 			}.bind(this),
100 			error: function(model, response) {
101 				console.warn('%s: Error loading model', this.toString());
102 				
103 				this.trigger('model:loadFailed', {
104 					response: response
105 				});
106 			}.bind(this)
107 		});
108 		
109 		return this;
110 	},
111 	
112 	/**
113 	 * Load a Backbone.Model directly or create a model from data
114 	 *
115 	 * @param {mixed} modelOrData	Backbone.Model to load or Object with data to create model from
116 	 *
117 	 * @returns {F.ModelComponent}	this, chainable
118 	 */
119 	load: function(modelOrData) {
120 		if (modelOrData instanceof Backbone.Model)
121 			this._setModel(modelOrData);
122 		else
123 			this._setModel(new this.Model(modelOrData));
124 		
125 		// Notify
126 		this.trigger('model:loaded', this.model);
127 		
128 		return this;
129 	},
130 	
131 	/**
132 	 * Save a model to the server
133 	 *
134 	 * @param {Object} data			Data to apply to model before performing save
135 	 * @param {Function} callback	Callback to execute on successful fetch
136 	 * @param {Object} options		Options to pass when calling model.save
137 	 *
138 	 * @returns {F.ModelComponent}	this, chainable
139 	 */
140 	save: function(data, callback, options) {
141 		if (this.model) {
142 			if (this.inDebugMode())
143 				console.log('%s: Saving...', this.toString());
144 			
145 			this.model.save(data || {}, _.extend({
146 				success: function(model, response) {
147 					if (this.inDebugMode())
148 						console.log('%s: Save successful', this.toString());
149 					
150 					if (typeof callback === 'function')
151 						callback.call(this, this.model, response);
152 						
153 					this.trigger('model:saved', {
154 						model: this.model,
155 						response: response
156 					});
157 				}.bind(this),
158 				error: function(model, response) {
159 					console.warn('%s: Error saving model', this.toString());
160 					
161 					this.trigger('model:saveFailed', {
162 						model: this.model,
163 						response: response
164 					});
165 				}.bind(this)
166 			}, options));
167 		}
168 		else {
169 			console.warn('%s: Cannot save, model is not truthy', this.toString());
170 		}
171 		return this;
172 	},
173 	
174 	/**
175 	 * Show this component, optionally fetching an item by ID or assigning a new model before render
176 	 *
177 	 * @param {Object} options			Show options
178 	 * @param {String} options.id		ID of model to fetch from the server before showing
179 	 * @param {Backbone.Model} options.model	Model to use directly (don't fetch)
180 	 *
181 	 * @returns {F.ModelComponent}	this, chainable
182 	 */
183 	show: function(options) {
184 		options = options || {};
185 		
186 		if (options.id) {
187 			if (this.inDebugMode()) {
188 				console.log('%s: fetching item with ID %s', this.toString(), options.id);
189 			}
190 			
191 			// Load the model by itemId, then show
192 			this.fetch(options.id, function(model) {
193 				if (this.inDebugMode()) {
194 					console.log('%s: fetch complete!', this.toString());
195 				}
196 				this.show({
197 					silent: options.silent
198 				}); // pass nothing to show and the view will re-render
199 			});
200 		}
201 		else if (options.model) {
202 			if (this.inDebugMode()) {
203 				console.log('%s: showing with new model', this.toString(), options.model);
204 			}
205 			
206 			this.load(options.model);
207 			this.show({
208 				silent: options.silent
209 			});
210 		}
211 		else
212 			this.inherited(arguments);
213 			
214 		return this;
215 	}

218 	 * Triggered after a successful save
219 	 *

226

231 	 * @event

233 	 * @param {Object} evt					Event object
234 	 * @param {Backbone.Model} evt.model	The model that failed to save
235 	 */
236

238 	 * Triggered when the model is loaded from the server or passed to load()
239 	 *
240 	 * @name F.ModelComponent#model:loaded
241 	 * @event
242 	 *	
243 	 * @param {Object} evt					Event object
244 	 * @param {Backbone.Model} evt.model	The model that was loaded
245 	 */
