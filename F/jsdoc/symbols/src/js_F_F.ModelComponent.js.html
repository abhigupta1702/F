1 F.ModelComponent = new Class(/** @lends F.ModelComponent# */{
  2 	toString: 'ModelComponent',
  3 	extend: F.Component,
  4 
  5 	/**
  6 	 * A component that can load and render a model
  7 	 *
  8 	 * @constructs
  9 	 * @extends F.Component
 10 	 *
 11 	 * @param {Object} options	Options for this component
 12 	 * @param {Object} options.Model	Model class this component will be operating on. Sets this.Model
 13 	 *
 14 	 * @property {Backbone.Model} Model		The model class to operate on. Not an instance of a model, but the model class itself.
 15 	 */
 16 	construct: function(options) {
 17 		this.Model = this.Model || options.Model;
 18 	},
 19 	
 20 	Model: Backbone.Model,
 21 	
 22 	/***
 23 	 * Refresh the model
 24 	 *
 25 	 * @param {Function} Callback to call after successful refresh
 26 	 *
 27 	 * @returns {F.ModelComponent}	this, chainable
 28 	 */
 29 	refresh: function(callback) {
 30 		this.model.fetch({
 31 			success: function() {
 32 				this.trigger('modelLoaded');
 33 				
 34 				if (typeof callback === 'function')
 35 					callback.call(this, this.model);
 36 			}.bind(this)
 37 		});
 38 		
 39 		return this;
 40 	},
 41 	
 42 	/**
 43 	 * Use a different item model
 44 	 *
 45 	 * @param {Backbone.Model} model
 46 	 *
 47 	 * @returns {F.ModelComponent}	this, chainable
 48 	 */
 49 	_setModel: function(model) {
 50 		if (this.model) {
 51 			// Unsubscribe from old model's change event
 52 			this.model.off('change', this.render);
 53 		}
 54 		
 55 		this.model = model;
 56 		
 57 		if (this.view) {
 58 			// Tell the view to re-render the next time it loads
 59 			this.view.rendered = null;
 60 		}
 61 		
 62 		// Subscribe to new model's change event
 63 		this.model.on('change', this.render);
 64 		
 65 		return this;
 66 	},
 67 		
 68 	/**
 69 	 * Load an item's model by ID or by model
 70 	 *
 71 	 * @param {Function} itemIdOrModel	ID of the item to fetch or already fetched model
 72 	 * @param {Function} callback	Callback to execute on successful fetch
 73 	 *
 74 	 * @returns {F.ModelComponent}	this, chainable
 75 	 */
 76 	load: function(itemIdOrModel, callback) {
 77 		// Load models 
 78 		if (typeof itemIdOrModel === 'string' || typeof itemIdOrModel === 'number') {
 79 			// Create a blank model
 80 			var data = {};
 81 			data[this.Model.prototype.idAttribute] = itemIdOrModel;
 82 			var model = new this.Model(data);
 83 		
 84 			// Fetch model contents
 85 			model.fetch({
 86 				// TBD: add fetch options
 87 				success: function() {
 88 					// Assign the model to the view
 89 					this._setModel(model);
 90 					
 91 					// Noftify
 92 					this.trigger('modelLoaded');
 93 					
 94 					// Call callback
 95 					if (typeof callback === 'function')
 96 						callback.call(this, model);
 97 				}.bind(this)
 98 			});
 99 		}
100 		else {
101 			// It must be an object
102 			this._setModel(itemIdOrModel);
103 		}
104 		
105 		return this;
106 	},
107 	
108 	/**
109 	 * Save a model to the server
110 	 *
111 	 * @param {Function} callback	Callback to execute on successful fetch
112 	 *
113 	 * @returns {F.ModelComponent}	this, chainable
114 	 */
115 	save: function(data, callback) {
116 		if (this.model) {
117 			if (F.options.debug)
118 				console.log('%s: Saving...', this.toString());
119 			
120 			this.model.save(data || {}, {
121 				success: function() {
122 					if (F.options.debug)
123 						console.log('%s: Save successful', this.toString());
124 					
125 					if (typeof callback === 'function')
126 						callback.call(this, this.model);
127 						
128 					this.trigger('saved', this.model);
129 				}.bind(this),
130 				error: function() {
131 					// TBD: add meaningful data to event properties
132 					console.warn('%s: Error saving model', this.toString());
133 					
134 					this.trigger('saveFailed', this.model);
135 				}.bind(this)
136 			});
137 		}
138 		else {
139 			console.warn('%s: Cannot save, model is not truthy', this.toString());
140 		}
141 		return this;
142 	},
143 	
144 	/**
145 	 * Show this component, optionally fetching an item by ID or assiging a new model before render
146 	 *
147 	 * @param {Object} options
148 	 *
149 	 * @returns {F.ModelComponent}	this, chainable
150 	 */
151 	show: function(options) {
152 		options = options || {};
153 		
154 		if (options.id) {
155 			if (F.options.debug) {
156 				console.log('%s: fetching item with ID %s', this.toString(), options.id);
157 			}
158 			
159 			// Load the model by itemId, then show
160 			this.load(options.id, function(model) {
161 				if (F.options.debug) {
162 					console.log('%s: fetch complete!', this.toString());
163 				}
164 				this.show(); // pass nothing to show and the view will re-render
165 			});
166 		}
167 		else if (options.model) {
168 			console.log('%s: showing with new model', this.toString(), options.model);
169 			this.load(options.model);
170 			this.show();
171 		}
172 		else
173 			this.inherited(arguments);
174 			
175 		return this;
176 	}
177 });
