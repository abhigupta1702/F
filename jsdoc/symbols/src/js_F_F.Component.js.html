1 (function() {
  2 	function decapitalize(str) {
  3 		return str.slice(0, 1).toLowerCase()+str.slice(1);	
  4 	}
  5 	
  6 	/**
  7 	 * Generic component class
  8 	 *
  9 	 * @class
 10 	 * @extends F.EventEmitter
 11 	 */
 12 	F.Component = new Class({
 13 		toString: 'Component',
 14 		extend: F.EventEmitter,
 15 		
 16 		/** @constructor */
 17 		construct: function(config) {
 18 			config = $.extend({}, {
 19 				singly: true,
 20 				visible: false
 21 			}, this.config || {}, config);
 22 		
 23 			// Sub components
 24 			this.components = {};
 25 		
 26 			// Show only one subcomponent at a time
 27 			this.singly = config.singly;
 28 		
 29 			// Visible or not
 30 			this.visible = config.visible;
 31 		
 32 			// Make sure the following functions are always called in scope
 33 			this.bind(this._setCurrentComponent); // shorthand for this._setCurrentComponent = this._setCurrentComponent.bind(this);
 34 			this.bind(this.render);
 35 		
 36 			if (this.visible) {
 37 				// Show the component once the call chain has returned
 38 				_.defer(function() {
 39 					this.show({
 40 						silent: true
 41 					});
 42 				}.bind(this));
 43 			}
 44 			else {
 45 				// Hide the component once the call chain has returned
 46 				_.defer(function() {
 47 					if (this.view && this.view.el) {
 48 						this.view.$el.hide();
 49 					}
 50 				}.bind(this));
 51 			}
 52 		},
 53 		
 54 		/** @lends F.Component# */
 55 		
 56 		/**
 57 		 * Destroy this instance and free associated memory
 58 		 */
 59 		destruct: function() {
 60 			// Destroy sub-components
 61 			for (var component in this.components) {
 62 				this.components[component].destruct();
 63 				delete this[component];
 64 			}
 65 		
 66 			// Clear references to components
 67 			delete this.components;
 68 		},
 69 		
 70 		/**
 71 		 * Render the view associated with this component, if it has one
 72 		 *
 73 		 */
 74 		render: function() {
 75 			if (this.view)
 76 				this.view.render();
 77 		},
 78 	
 79 		/**
 80 		 * Add an instance of another component as a sub-component.
 81 		 *
 82 		 *  this[subComponent.toString()] is used to reference the sub-component:
 83 		 *  		
 84 		 *  		this.List.show();
 85 		 *  		
 86 		 *  You can give a component an optional custom name as the second argument, then reference as such:
 87 		 *  		
 88 		 *  		this.myCustomComponent.show();
 89 		 *  		
 90 		 *
 91 		 * @param {F.Component} component	Instance of component
 92 		 * @param {Function} componentName	Optional custom name for this component
 93 		 *
 94 		 * @returns {F.Component}	The sub-component that was added
 95 		 */
 96 		addComponent: function(component, componentName) {
 97 			// Tell component its new name, if provided
 98 			if (componentName)
 99 				component.setName(componentName);
100 		
101 			// Get the name of the component
102 			componentName = decapitalize(component.toString());
103 		
104 			// Store component
105 			this[componentName] = this.components[componentName] = component;
106 		
107 			// Hide view by default
108 			if (component.view) {
109 				if (component.view.el) {
110 					component.view.$el.hide();
111 				}
112 				else {
113 					console.warn('Component %s has a view without an element', componentName, component, component.view, component.view.options);
114 				}
115 			}
116 		
117 			// Show a sub-component when it shows one of it's sub-components
118 			component.on('component:shown', this._setCurrentComponent);
119 			
120 			return component;
121 		},
122 	
123 		/**
124 		 * Remove a sub-component
125 		 *
126 		 * @param {Function} componentName	Component name
127 		 *
128 		 * @returns {F.Component}	this, chainable
129 		 */
130 		removeComponent: function(componentName) {
131 			var component = this[componentName];
132 		
133 			if (component !== undefined) {
134 				component.off('component:shown', this._componentShowHandler);
135 		
136 				delete this[componentName];
137 				delete this.components[componentName];
138 			}
139 		
140 			return this;
141 		},
142 	
143 	
144 		/**
145 		 * Handles showing/hiding components in singly mode
146 		 *
147 		 * @param {Function} componentName	Component name
148 		 */
149 		_setCurrentComponent: function(componentName) {
150 			var newComponent = this.components[componentName];
151 		
152 			if (newComponent !== undefined) {
153 				// hide current component(s) for non-overlays
154 				if (this.singly && !newComponent.overlay) {
155 					this.hideComponents();
156 				}
157 			
158 				// Show self
159 				// this.show(); // should not be necessary?
160 			}
161 		},
162 	
163 		/**
164 		 * Show this component and emit an event so parent components can show themselves. Use options.silent to prevent component:shown event from firing
165 		 *
166 		 * @param {Object} options	Options object
167 		 *
168 		 * @returns {F.Component}	this, chainable
169 		 */
170 		show: function(options) {
171 			options = options || {};
172 		
173 			// Debug output
174 			if (F.config.debug) {
175 				// Don't show if already shown
176 				if (this.visible) {		
177 					console.log('%s: not showing self; already visible', this.toString());
178 				}
179 				else
180 					console.log('%s: showing self', this.toString());
181 			}
182 		
183 			if (!this.visible && !options.silent) {
184 				// Always trigger event before we show ourself so others can hide/show
185 				this.trigger('component:shown', this.toString(), this);	
186 			}
187 		
188 			// Always call show on the view so it has a chance to re-render
189 			if (this.view) {
190 				this.view.show();
191 			}
192 		
193 			this.visible = true;
194 	
195 			return this;
196 		},
197 	
198 		/**
199 		 * Hide this component
200 		 *
201 		 * @returns {F.Component}	this, chainable
202 		 */
203 		hide: function() {
204 			if (!this.visible)
205 				return false;
206 		
207 			if (F.config.debug) {
208 				console.log('%s: hiding self', this.toString());
209 			}
210 			
211 			// Hide the view
212 			if (this.view)
213 				this.view.hide();
214 			
215 			// Trigger event after we hide ourself so we're out of the way before the next action
216 			this.trigger('component:hidden', this.toString(), this);
217 		
218 			this.visible = false;
219 	
220 			return this;
221 		},
222 	
223 		/**
224 		 * Check if this component is currently visible
225 		 *
226 		 * @returns {Boolean} Component is visible
227 		 */
228 		isVisible: function() {
229 			return this.visible;
230 		},
231 	
232 		/**
233 		 * Set a custom name for this component. Only useful before passing to addComponent
234 		 *
235 		 * @param {Function} componentName	Component name
236 		 *
237 		 * @returns {F.Component}	this, chainable
238 		 */
239 		setName: function(customName) {
240 			this.toString = function() {
241 				return customName;
242 			};
243 			
244 			return this;
245 		},
246 	
247 		/**
248 		 * Hide all sub-components
249 		 *
250 		 */
251 		hideComponents: function() {
252 			for (var componentName in this.components) {
253 				this.hideComponent(componentName);
254 			}
255 		
256 			return this;
257 		},
258 		
259 		/**
260 		 * Hide a sub-component of this component by name. Only useful if config.singly is false
261 		 *
262 		 * @param {Function} componentName	Component name
263 		 *
264 		 * @returns {F.Component}	this, chainable
265 		 */
266 		hideComponent: function(componentName) {
267 			var component = this.components[componentName];
268 			if (component !== undefined) {
269 				if (component.isVisible()) {
270 					// hide the component's element
271 					component.hide();
272 				}
273 			}
274 			else {
275 				console.warn(this.toString()+': cannot hide component %s, component not found', componentName);
276 			}
277 		
278 			return this;
279 		},
280 	
281 		/**
282 		 * Show a sub-component of this component by name.
283 		 *
284 		 * @param {Function} componentName	Component name
285 		 *
286 		 * @returns {F.Component}	this, chainable
287 		 */
288 		showComponent: function(componentName) {
289 			// Show the sub section, if not already showing
290 			var newComponent = this.components[componentName];
291 		
292 			if (newComponent !== undefined) {
293 				if (!newComponent.isVisible()) {
294 					// Hide the old component and show ourselves
295 					this._setCurrentComponent(componentName);
296 			
297 					// Show new component
298 					newComponent.show();
299 				}
300 				else {
301 					console.log(this.toString()+': not showing component %s, already visible', componentName);
302 				}
303 			}
304 			else {
305 				console.warn(this.toString()+': Cannot show component "'+componentName+'", not found');
306 			}
307 		
308 			return this;
309 		}
310 	});
311 }());
